== Installation

=== Maven
[source,xml]
----
<dependency>
    <groupId>com.wilsonfranca</groupId>
    <artifactId>dynamodb-client-starter</artifactId>
    <version>1.0.0-SNAPSHOT</version>
</dependency>
----
=== Gradle Groovy
[source,groovy]
----
implementation 'com.wilsonfranca:dynamodb-client-starter:1.0.0-SNAPSHOT'
----
=== Gradle Kotlin
[source,kotlin]
----
implementation("com.wilsonfranca:dynamodb-client-starter:1.0.0-SNAPSHOT")
----
